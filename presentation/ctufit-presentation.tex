\section{Zadání}
\begin{frame}
    \frametitle{Zadání}
    \begin{itemize}
        \item návrh a implementace webové aplikace prezentující aktivitu vývojářů
        \item analýza metrik a webových API systémů GitHub a GitLab
        \item konsolidace dat z VCS a párování uživatelských účtů
        \item vizualizace metrik pomocí grafů
        \item definice výzev a získávání „odznáčků“ za aktivitu
        \item ověření implementovaného řešení uživatelským testováním
    \end{itemize}
\end{frame}

\section{Analýza}
\begin{frame}
    \frametitle{Analýza}
    \begin{itemize}
        \item \textbf{Porovnání VCS platforem:} GitHub a GitLab API (REST vs. GraphQL)
        \item \textbf{Metodiky:} SPACE framework a DORA metriky
        \item \textbf{Motivace vývojářů a gamifikace}
        \item \textbf{Uživatelské požadavky:} funkční a nefunkční požadavky
        \item \textbf{Existující řešení:} Swarmia a LinearB
    \end{itemize}
\end{frame}

\subsection{SPACE framework}
\begin{frame}
    \frametitle{Analýza}
    \framesubtitle{SPACE framework}
    \begin{itemize}
        \item pomáhá porozumět produktivitě vývojářů pomocí různorodých aspektů
        \item navržen zástupci univerzitní a komerční sféry
        \item definuje 5 klíčových dimenzí
        \item doporučeno sledovat metriky z minimálně 3 různých dimenzí
        \item na jeho základě byly vybrány metriky, které lze získat z VCS platforem
    \end{itemize}
\end{frame}

\section{Architektura a technologie}
\begin{frame}
    \frametitle{Architektura a technologie}
    \begin{itemize}
        \item \textbf{Architektura:}
        \begin{itemize}
            \item monolitická architektura, návrhový vzor MVC
            \item Single Page Application
        \end{itemize}
        \item \textbf{Backend:} PHP 8, Laravel framework
        \item \textbf{Frontend:} Vue.js, Highcharts
        \item \textbf{Propojení vrstev:} knihovna Inertia.js
        \item \textbf{Nasazení:} kontejnerizace pomocí technologie Docker
    \end{itemize}
\end{frame}
